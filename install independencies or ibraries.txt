CREATE TABLE AccessLevels (
    AccessLevelId INT IDENTITY(1,1) PRIMARY KEY,
    AccessLevelName NVARCHAR(50) NOT NULL UNIQUE,
    DisplayName NVARCHAR(100) NOT NULL,
    Description NVARCHAR(255) NULL,
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME NOT NULL DEFAULT GETDATE()

ALTER TABLE AccessLevels
ADD 
    SortOrder INT NOT NULL CONSTRAINT DF_AccessLevels_SortOrder DEFAULT 0,
    IsSystemRole BIT NOT NULL CONSTRAINT DF_AccessLevels_IsSystemRole DEFAULT 0,
    UpdatedAt DATETIME NULL;


-- ============================================
-- Update default role ordering and mark system roles
-- ============================================

UPDATE AccessLevels
SET 
    SortOrder = CASE AccessLevelName
        WHEN 'TeacherLevel' THEN 1
        WHEN 'HODLevel' THEN 2
        WHEN 'HOSLevel' THEN 3
        WHEN 'PrintingLevel' THEN 4
        WHEN 'AdminLevel' THEN 5
        WHEN 'SuperAdminLevel' THEN 6
        WHEN 'RequestOnlyLevel' THEN 7
        WHEN 'ViewOnlyLevel' THEN 8
        ELSE 99
    END,
    IsSystemRole = CASE 
        WHEN AccessLevelName IN (
            'TeacherLevel',
            'HODLevel',
            'HOSLevel',
            'PrintingLevel',
            'AdminLevel',
            'SuperAdminLevel'
        ) THEN 1
        ELSE 0
    END,
    UpdatedAt = GETDATE();


CREATE TABLE Roles (
    RoleId INT IDENTITY(1,1) PRIMARY KEY,
    RoleName NVARCHAR(50) NOT NULL UNIQUE,
    DisplayName NVARCHAR(100) NOT NULL,
    AccessLevelId INT NOT NULL,
    IsSystemRole BIT NOT NULL DEFAULT 0,
    IsActive BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT FK_Roles_AccessLevels
        FOREIGN KEY (AccessLevelId)
        REFERENCES AccessLevels(AccessLevelId)
);
